# 登录

> 需要登录访问项目

在 `main.js` 中

```js
// 引入路由
import router from './router'

// 登录
router.beforeEach((to, from, next) => {
  // 看是否已经登录
  const isLogin = window.sessionStorage.getItem('isLogin')
  // 未登录则跳转登录
  if (!isLogin && to.name !== 'login') {
    next({
      path: '/',
      query: {
        from: to.fullPath
      }
    })
  } else {
    next()
  }
})

new Vue({
  el: '#app',
  router, // 路由
  store,
  components: {
    App
  },
  template: '<App/>'
})
```
